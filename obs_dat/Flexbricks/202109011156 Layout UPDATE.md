# Multiple updates in one query:

```ad-info
title: Example 1
collapse: open
```sql
UPDATE sUILayoutField lf
SET lookupqueryid = CASE lf.Name 
				WHEN 'DivisionID' 
					THEN '8b5ad40a-1ef5-a81e-8c96-9215fe5fa339' 
				WHEN 'StyleType' 
					THEN  'c9b16bbb-a03a-3ea4-f24f-04183c514cb7' 
		          ELSE lookupqueryid
		          END
FROM sUILayout l	                  
WHERE LOWER(l.LayoutName) = LOWER('Download_TechPack_Search_SRMUSER') 
	 AND LOWER(lf.Name) IN (LOWER('StyleType'), LOWER('DivisionID'))
	 AND l.LayoutID = lf.LayoutID;
```

It is the same as:

```ad-info
title: Example 1.1
collapse: open
```sql
UPDATE sUILayoutField lf
SET lookupqueryid = '8b5ad40a-1ef5-a81e-8c96-9215fe5fa339'
FROM sUILayout l
WHERE LOWER(l.LayoutName) = LOWER('Download_TechPack_Search_SRMUSER') AND LOWER(lf.Name) = LOWER('DivisionID')
AND l.LayoutID = lf.LayoutID;

UPDATE sUILayoutField lf
SET lookupqueryid = 'c9b16bbb-a03a-3ea4-f24f-04183c514cb7'
FROM sUILayout l
WHERE LOWER(l.LayoutName) = LOWER('Download_TechPack_Search_SRMUSER') AND LOWER(lf.Name) = LOWER('StyleType')
AND l.LayoutID = lf.LayoutID;
```
